# Statistical significance
# Choose the same number of longs and shorts but randomise where they occur
# What is the prob of this doing as well as the strategy did
test_stat = train.results()['Strategy']
end = []
market = train.market_ret()
stance = train.stance()
for i in np.arange(10000):
    stance = pd.Series(np.random.choice(stance, size = len(stance), replace = False), index = stance.index)
    strategy = market * stance.shift(1) 
    years = len(market) / 365.25
    ret = ((np.exp(strategy.cumsum()[-1]) ** (1 / years) - 1) * 100)
    end.append(ret)
